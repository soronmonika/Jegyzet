var http = require("http");
const hostname = "127.0.0.1"; //A szerver IP cím(LOCALHOST)
const port = 3000; //A szerver potja

const server = http.createServer((req, res) => {
    res.statusCode = 200;//Státuszkód szerver felé, 200 = minden rendben -404
    res.setHeader("Content-Type", "text/plain");//Tartalom típus
    res.end("Szeretem a NodeJS");//Üzenet átadása az oldalra, és kapcsolat lezárása});
server.listen(port, hostname, () => {
    console.log(`A szerverünk fut a http://${hostname}:${port} címen`);// http://localhost:3000   ||| http://127.0.0.1:3000
})
